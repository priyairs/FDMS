@startuml
title Hospital Class Diagram

class Hospital {
  - name: String
  - address: String
  + addDepartment(d: Department)
}

class Department {
  - name: String
  + addDoctor(d: Doctor)
  + addPatient(p: Patient)
}

class Person {
  - id: int
  - name: String
  - dob: Date
}

class Doctor {
  - specialization: String
  - licenseNo: String
  + diagnose(p: Patient)
}

class Patient {
  - medicalRecordNo: String
  - allergies: String[]
  + getHistory(): History
}

class Appointment {
  - apptId: int
  - dateTime: DateTime
  - status: String
  + reschedule(dt: DateTime)
}

class MedicalRecord {
  - recordId: int
  - entries: List<Entry)
  + addEntry(e: Entry)
}

Person <|-- Doctor
Person <|-- Patient

MedicalRecord "1" o-- "*" Entry

@enduml


@startuml
title College Management System - Class Diagram

class College {
  - name: String
  - address: String
  + addDepartment(d: Department)
}

class Department {
  - name: String
  + addCourse(c: Course)
  + addStaff(s: Staff)
}

class Person {
  - id: int
  - name: String
  - email: String
}

class Student {
  - rollNo: String
  - courses: List<Course>
  + enroll(course: Course)
}

class Staff {
  - staffId: String
  - role: String
}

class Course {
  - code: String
  - title: String
  - credits: int
  + assignInstructor(staff: Staff)
}

class Examination {
  - examId: int
  - date: Date
  + schedule()
}

Person <|-- Student
Person <|-- Staff
College "1" o-- "many" Department
Department "1" o-- "many" Course
Department "1" o-- "many" Staff
Student "many" -- "many" Course : enrolls
Course "1" -- "many" Examination

@enduml
